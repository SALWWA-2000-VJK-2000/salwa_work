        SET loanDuration TO ld

CLASS JournalIssue INHERITS ReferenceTextbook

    CONSTRUCTOR(t : TEXT, b : TEXT, ld : NUMBER, cr : ConditionReport, hfpd : NUMBER)

CLASS StudentAccount

        SET name TO n
        SET location TO l

CLASS RollerCoaster INHERITS Ride
    ATTRIBUTES
        carCapacity : NUMBER
        cycleTime : NUMBER

    CONSTRUCTOR(n : TEXT, l : TEXT, log : SafetyInspectionLog, cc : NUMBER, ct : NUMBER)
        SUPER(n, l, log)
        SET carCapacity TO cc
        SET cycleTime TO ct
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_WAIT_TIME(peopleInLine : NUMBER)
            SET trainsNeeded TO CEIL(peopleInLine / carCapacity)
            RETURN trainsNeeded * cycleTime
        END METHOD
END CLASS

CLASS WaterRide INHERITS Ride
    ATTRIBUTES
        boatCapacity : NUMBER
        cycleTime : NUMBER
        loadingDelay : NUMBER

    CONSTRUCTOR(n : TEXT, l : TEXT, log : SafetyInspectionLog, bc : NUMBER, ct : NUMBER, ld : NUMBER)
        SUPER(n, l, log)
        SET boatCapacity TO bc
        SET cycleTime TO ct
        SET loadingDelay TO ld
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_WAIT_TIME(peopleInLine : NUMBER)
            SET launchesNeeded TO CEIL(peopleInLine / boatCapacity)
            RETURN launchesNeeded * (cycleTime + loadingDelay)
        END METHOD
END CLASS

CLASS VIPAccessRide INHERITS RollerCoaster
    METHODS
        OVERRIDE CALCULATE_WAIT_TIME(peopleInLine : NUMBER, vipGuests : NUMBER)
            SET standardGuests TO peopleInLine - vipGuests
            SET standardWait TO SUPER.CALCULATE_WAIT_TIME(standardGuests)
            SET vipWait TO cycleTime
            DISPLAY "VIP Wait Time = " + vipWait
            DISPLAY "Standard Wait Time = " + standardWait
            RETURN 0
        END METHOD
END CLASS

CLASS FixedPriceInvoice INHERITS Invoice
    ATTRIBUTES
        projectFee : NUMBER

    CONSTRUCTOR(inum : TEXT, di : TEXT, cn : TEXT, pt : PaymentTerms, pf : NUMBER)
        SUPER(inum, di, cn, pt)
        SET projectFee TO pf
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_TOTAL()
            RETURN projectFee
        END METHOD
END CLASS


CLASS HourlyInvoice INHERITS Invoice
    ATTRIBUTES
        hours : LIST<NUMBER>
        approvals : LIST<BOOLEAN>
        hourlyRate : NUMBER

    CONSTRUCTOR(inum : TEXT, di : TEXT, cn : TEXT, pt : PaymentTerms, hr : NUMBER)
        SUPER(inum, di, cn, pt)
        SET hours TO LIST []
        SET approvals TO LIST []
        SET hourlyRate TO hr
    END CONSTRUCTOR

    METHODS
        ADD_HOUR(h : NUMBER, approved : BOOLEAN)
            ADD h TO hours
            ADD approved TO approvals
        END METHOD

        OVERRIDE CALCULATE_TOTAL()
            SET total TO 0
            SET i TO 0
            REPEAT WHILE i LESS THAN SIZE OF hours
                IF approvals[i] EQUALS TRUE THEN
                    SET total TO total + (hours[i] * hourlyRate)
                END IF
                SET i TO i + 1
            END REPEAT
            RETURN total
        END METHOD
END CLASS

CLASS RetainerInvoice INHERITS FixedPriceInvoice
    ATTRIBUTES
        includedHours : NUMBER
        extraHourlyRate : NUMBER
        extraHours : NUMBER

    CONSTRUCTOR(inum : TEXT, di : TEXT, cn : TEXT, pt : PaymentTerms, pf : NUMBER, ih : NUMBER, ehr : NUMBER)
        SUPER(inum, di, cn, pt, pf)
        SET includedHours TO ih
        SET extraHourlyRate TO ehr
        SET extraHours TO 0
    END CONSTRUCTOR

    METHODS
        ADD_EXTRA_HOURS(h : NUMBER)
            SET extraHours TO extraHours + h
        END METHOD

        OVERRIDE CALCULATE_TOTAL()
            SET baseFee TO SUPER.CALCULATE_TOTAL()
            SET extraFee TO extraHours * extraHourlyRate
            RETURN baseFee + extraFee
        END METHOD
END CLASS

CLASS Server
    ATTRIBUTES
        serialNumber : TEXT
        uptime : NUMBER
        hardware : HardwareSpecification

    CONSTRUCTOR(sn : TEXT, up : NUMBER, hw : HardwareSpecification)
        SET serialNumber TO sn
        SET uptime TO up
        SET hardware TO hw
    END CONSTRUCTOR

    METHODS
        CALCULATE_LOAD()
            RETURN 0
        END METHOD

        DISPLAY_SERVER()
            DISPLAY serialNumber
            DISPLAY uptime
            hardware.VIEW_SPEC()
        END METHOD
END CLASS

CLASS WebServer INHERITS Server
    ATTRIBUTES
        activeConnections : NUMBER
        maxConnections : NUMBER

    CONSTRUCTOR(sn : TEXT, up : NUMBER, hw : HardwareSpecification, ac : NUMBER, mc : NUMBER)
        SUPER(sn, up, hw)
        SET activeConnections TO ac
        SET maxConnections TO mc
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_LOAD()
            RETURN (activeConnections / maxConnections) * 100
        END METHOD
END CLASS

CLASS DatabaseServer INHERITS Server
    ATTRIBUTES
        queryThroughput : NUMBER
        maxQueriesPerSecond : NUMBER

    CONSTRUCTOR(sn : TEXT, up : NUMBER, hw : HardwareSpecification, qt : NUMBER, mqs : NUMBER)
        SUPER(sn, up, hw)
        SET queryThroughput TO qt
        SET maxQueriesPerSecond TO mqs
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_LOAD()
            IF queryThroughput >= maxQueriesPerSecond THEN
                DISPLAY "Server at full capacity, no new connections allowed"
                RETURN 100
            ELSE
                RETURN (queryThroughput / maxQueriesPerSecond) * 100
            END IF
        END METHOD
END CLASS

CLASS CacheServer INHERITS DatabaseServer
    ATTRIBUTES
        cacheHitRatio : NUMBER

    CONSTRUCTOR(sn : TEXT, up : NUMBER, hw : HardwareSpecification, qt : NUMBER, mqs : NUMBER, chr : NUMBER)
        SUPER(sn, up, hw, qt, mqs)
        SET cacheHitRatio TO chr
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_LOAD()
            SET baseLoad TO SUPER.CALCULATE_LOAD()
            SET adjustedLoad TO baseLoad * (1 - (cacheHitRatio / 100))
            RETURN adjustedLoad
        END METHOD
END CLASS

        END METHOD
END CLASS
