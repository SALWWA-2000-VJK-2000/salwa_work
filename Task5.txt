//Task 293: Shape Drawing App  
CLASS Shape
	ATTRIBUTES
		color: TEXT
		position: TEXT 
	
	CONSTRUCTOR(c: TEXT, p:TEXT)
		SET color TO c
		SET position TO p 
		
	END CONSTRUCTOR 
	
	METHODS
        CALCULATE_AREA()
            DISPLAY " shape area"
        END METHOD

        CALCULATE_PERIMETER()
            DISPLAY " shape perimeter"
        END METHOD
END CLASS


CLASS Circle INHERITS Shape 
	ATTRIBUTES
		radius: NUMBER
	CONSTRUCTOR(c : TEXT, p : TEXT, r : NUMBER)
        SUPER(c, p)
        SET radius TO r
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_AREA()
            RETURN 3.14 * radius * radius
        END METHOD

        OVERRIDE CALCULATE_PERIMETER()
            RETURN 2 * 3.14 * radius
        END METHOD
END CLASS


CLASS Rectangle INHERITS Shape 
	ATTRIBUTES
		side1: NUMBER
		side2: NUMBER
		side3: NUMBER
		height : NUMBER 
		
	CONSTRUCTOR(c : TEXT, p : TEXT, s1: NUMBER, s2: NUMBER, s3: NUMBER, h: NUMBER)
        SUPER(c, p)
        SET side1 TO s1
        SET side2 TO s2
		SET side3 TO s3
		SET height TO h
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_AREA()
            RETURN (1 / 2) * sideA * height
        END METHOD
		
		OVERRIDE CALCULATE_PERIMETER()
            RETURN sideA + sideB + sideC
        END METHOD

END CLASS


CLASS Rectangle INHERITS Shape 
	ATTRIBUTES
		width: NUMBER
		height: NUMBER
		
	CONSTRUCTOR(c : TEXT, p : TEXT, w: NUMBER, h: NUMBER)
        SUPER(c, p)
        SET width TO w
		SET height TO h
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_AREA()
            RETURN width * height
        END METHOD

        OVERRIDE CALCULATE_PERIMETER()
            RETURN 2(width *height)
        END METHOD
END CLASS

CLASS PaymentMethod
    ATTRIBUTES
        amount : NUMBER
        status : TEXT   

    CONSTRUCTOR(a : NUMBER)
        SET amount TO a
        SET status TO "pending"
    END CONSTRUCTOR

    METHODS
        GET_AMOUNT()
            RETURN amount
        END METHOD

        GET_STATUS()
            RETURN status
        END METHOD

        PROCESS_PAYMENT()
            DISPLAY " payment processing"
            RETURN FALSE
        END METHOD
END CLASS


CLASS CreditCardPayment INHERITS PaymentMethod
    ATTRIBUTES
        cardNumber : TEXT
        expiryDate : TEXT

    CONSTRUCTOR(a : NUMBER, cn : TEXT, ed : TEXT)
        SUPER(a)
        SET cardNumber TO cn
        SET expiryDate TO ed
    END CONSTRUCTOR

    METHODS
        OVERRIDE PROCESS_PAYMENT()
            IF LENGTH(cardNumber) EQUALS 16 AND expiryDate NOT EMPTY THEN
                SET status TO "completed"
                DISPLAY "Credit card payment approved for " + amount
                RETURN TRUE
            ELSE
                DISPLAY "Credit card validation failed"
                RETURN FALSE
            END IF
        END METHOD
END CLASS


CLASS BankTransferPayment INHERITS PaymentMethod
    ATTRIBUTES
        accountNumber : TEXT

    CONSTRUCTOR(a : NUMBER, acc : TEXT)
        SUPER(a)
        SET accountNumber TO acc
    END CONSTRUCTOR

    METHODS
        OVERRIDE PROCESS_PAYMENT()
            IF LENGTH(accountNumber) GREATER THAN 5 THEN
                DISPLAY "Processing bank transfer... please wait"
                SET status TO "completed"
                DISPLAY "Bank transfer completed for " + amount
                RETURN TRUE
            ELSE
                DISPLAY "Invalid account number"
                RETURN FALSE
            END IF
        END METHOD
END CLASS


CLASS WalletPayment INHERITS PaymentMethod
    ATTRIBUTES
        walletBalance : NUMBER

    CONSTRUCTOR(a : NUMBER, balance : NUMBER)
        SUPER(a)
        SET walletBalance TO balance
    END CONSTRUCTOR

    METHODS
        OVERRIDE PROCESS_PAYMENT()
            IF walletBalance GREATER THAN OR EQUAL TO amount THEN
                SET walletBalance TO walletBalance - amount
                SET status TO "completed"
                DISPLAY "Wallet payment completed for " + amount
                RETURN TRUE
            ELSE
                DISPLAY "Insufficient wallet balance"
                RETURN FALSE
            END IF
        END METHOD
END CLASS

CLASS Order
    ATTRIBUTES
        items : LIST<Item>
        payment : PaymentMethod

    CONSTRUCTOR()
        SET items TO LIST []
        SET payment TO NULL
    END CONSTRUCTOR

    METHODS
        ADD_ITEM(i : Item)
            ADD i TO items
        END METHOD

        CALCULATE_TOTAL()
            SET total TO 0
            SET i TO 0
            REPEAT WHILE i LESS THAN SIZE OF items
                SET total TO total + items[i].GET_PRICE()
                SET i TO i + 1
            END REPEAT
            RETURN total
        END METHOD

        CHECKOUT(pm : PaymentMethod)
            SET payment TO pm
            DISPLAY "Checkout total: " + CALCULATE_TOTAL()
            IF payment.PROCESS_PAYMENT() EQUALS TRUE THEN
                DISPLAY "Order payment successful"
            ELSE
                DISPLAY "Order payment failed"
            END IF
        END METHOD
END CLASS

CLASS CreditCard INHERITS Card
	ATTRIBUTES
		creditLimit : NUMBER
        outstandingBalance : NUMBER
		
	CONSTRUCTOR(cn: NUMBER, ed: TEXT, s:TEXE, c: Chip, cl: NUMBER)
		SUPER(cn, ed, s, c)
		SET creditLimit TO cl
		SET outstandingBalance TO 0
	END CONSTRUCTOR
	
	METHODS
		OVERRIDE AUTHORISE_PURCHASE(amount: NUMBER)
			IF status EQUALS "blocked"
				DISPLAY " CARD BLOCKED "
				RETURN FALSE
			END IF 
			IF chip.VALIDATE_SECURITY() EQUALS TRUE AND (outstandingBalance + amount) <= creditLimit THEN
                SET outstandingBalance TO outstandingBalance + amount
				DISPLAY amount
				RETURN TRUE
			ELSE
				DISPLAY "Credit limit exceeded"
				RETURN FALSE
			END IF 
		END METHOD
		
		GET_OUTSTANDING_BALANCE()
            RETURN outstandingBalance
        END METHOD
END CLASS

START

CLASS Shift
    ATTRIBUTES
        startTime : TEXT
        endTime : TEXT
        ward : TEXT

    CONSTRUCTOR(st : TEXT, et : TEXT, w : TEXT)
        SET startTime TO st
        SET endTime TO et
        SET ward TO w
    END CONSTRUCTOR

    METHODS
        DESCRIBE()
            DISPLAY "Shift in " + ward + " from " + startTime + " to " + endTime
        END METHOD
END CLASS


CLASS Ward
    ATTRIBUTES
        name : TEXT
        shifts : LIST<Shift>

    CONSTRUCTOR(n : TEXT)
        SET name TO n
        SET shifts TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_SHIFT(s : Shift)
            ADD s TO shifts
        END METHOD

        VIEW_SHIFTS()
            SET i TO 0
            REPEAT WHILE i LESS THAN SIZE OF shifts
                shifts[i].DESCRIBE()
                SET i TO i + 1
            END REPEAT
        END METHOD
END CLASS

        SET author TO a
        SET totalPages TO tp

CLASS Novel INHERITS Publication
    ATTRIBUTES
        pagesRead : NUMBER

    CONSTRUCTOR(t : TEXT, a : TEXT, tp : NUMBER, ci : CoverImage, pr : NUMBER)
        CALL SUPER(t, a, tp, ci)
        SET pagesRead TO pr

//Task 302: Smart Home Devices (Medium) 
CLASS Device
    ATTRIBUTES
        name : TEXT
        roomLocation : TEXT
        status : TEXT

    CONSTRUCTOR(n : TEXT, r : TEXT, st : TEXT)
        SET name TO n
        SET roomLocation TO r
        SET status TO st
    END CONSTRUCTOR

    METHODS
        TURN_ON()
            SET status TO "on"
        END METHOD

        TURN_OFF()
            SET status TO "off"
        END METHOD

        GET_STATUS()
            RETURN status
        END METHOD

        GET_POWER_USAGE()
            RETURN 0
        END METHOD
END CLASS


CLASS SmartLight INHERITS Device
    ATTRIBUTES
        brightness : NUMBER

    CONSTRUCTOR(n : TEXT, r : TEXT, st : TEXT, b : NUMBER)
        SUPER(n, r, st)
        SET this.brightness TO b
    END CONSTRUCTOR

    METHODS
        OVERRIDE GET_POWER_USAGE()
            IF status EQUALS "on" THEN
                RETURN brightness * 0.1
            ELSE
                RETURN 0
            END IF
        END METHOD
END CLASS


CLASS SmartThermostat INHERITS Device
    ATTRIBUTES
        targetTemperature : NUMBER
        currentTemperature : NUMBER

    CONSTRUCTOR(n : TEXT, r : TEXT, st : TEXT, target : NUMBER, current : NUMBER)
        SUPER(n, r, st)
        SET targetTemperature TO target
        SET currentTemperature TO current
    END CONSTRUCTOR

    METHODS
        OVERRIDE GET_POWER_USAGE()
            IF status EQUALS "on" THEN
                SET difference TO ABS(targetTemperature - currentTemperature)
                RETURN difference * 0.5
            ELSE
                RETURN 0
            END IF
        END METHOD
END CLASS


CLASS StorageUnit
    ATTRIBUTES
        capacity : NUMBER
        usedSpace : NUMBER

    CONSTRUCTOR(cap : NUMBER)
        SET capacity TO cap
        SET usedSpace TO 0
    END CONSTRUCTOR

    METHODS
        ADD_RECORDING(size : NUMBER)
            IF usedSpace + size LESS THAN OR EQUAL TO capacity THEN
                SET usedSpace TO usedSpace + size
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        GET_USED_SPACE()
            RETURN usedSpace
        END METHOD
END CLASS


CLASS SmartCamera INHERITS Device
    ATTRIBUTES
        recording : BOOLEAN
        storage : StorageUnit

    CONSTRUCTOR(n : TEXT, r : TEXT, st : TEXT, storageObj : StorageUnit)
        SUPER(n, r, st)
        SET recording TO FALSE
        SET storage TO storageObj
    END CONSTRUCTOR

    METHODS
        START_RECORDING(size : NUMBER)
            IF status EQUALS "on" THEN
                IF storage.ADD_RECORDING(size) EQUALS TRUE THEN
                    SET recording TO TRUE
                ELSE
                    SET recording TO FALSE
                END IF
            END IF
        END METHOD

        STOP_RECORDING()
            SET recording TO FALSE
        END METHOD

        OVERRIDE GET_POWER_USAGE()
            IF status EQUALS "on" AND recording EQUALS TRUE THEN
                RETURN 5
            ELSE IF status EQUALS "on" THEN
                RETURN 1
            ELSE
                RETURN 0
            END IF
        END METHOD
END CLASS


CLASS Room
    ATTRIBUTES
        name : TEXT
        devices : LIST<Device>

    CONSTRUCTOR(n : TEXT)
        SET name TO n
        SET devices TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_DEVICE(d : Device)
            ADD d TO devices
        END METHOD

        VIEW_DEVICES()
            SET i TO 0
            REPEAT WHILE i LESS THAN SIZE OF devices
                DISPLAY "Device: " + devices[i].name + 
                        " (" + devices[i].roomLocation + ") Status: " + devices[i].GET_STATUS()
                DISPLAY "Power Usage = " + devices[i].GET_POWER_USAGE()
                SET i TO i + 1
            END REPEAT
        END METHOD
END CLASS
