//Task 293: Shape Drawing App  
CLASS Shape
	ATTRIBUTES
		color: TEXT
		position: TEXT 
	
	CONSTRUCTOR(c: TEXT, p:TEXT)
		SET color TO c
		SET position TO p 
		
	END CONSTRUCTOR 
	
	METHODS
        CALCULATE_AREA()
            DISPLAY " shape area"
        END METHOD

        CALCULATE_PERIMETER()
            DISPLAY " shape perimeter"
        END METHOD
END CLASS


CLASS Circle INHERITS Shape 
	ATTRIBUTES
		radius: NUMBER
	CONSTRUCTOR(c : TEXT, p : TEXT, r : NUMBER)
        SUPER(c, p)
        SET radius TO r
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_AREA()
            RETURN 3.14 * radius * radius
        END METHOD

        OVERRIDE CALCULATE_PERIMETER()
            RETURN 2 * 3.14 * radius
        END METHOD
END CLASS


CLASS Triangle INHERITS Shape 
    ATTRIBUTES
        side1 : NUMBER
        side2 : NUMBER
        side3 : NUMBER
        height : NUMBER 

    CONSTRUCTOR(c : TEXT, p : TEXT, s1 : NUMBER, s2 : NUMBER, s3 : NUMBER, h : NUMBER)
        SUPER(c, p)
        SET side1 TO s1
        SET side2 TO s2
        SET side3 TO s3
        SET height TO h
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_AREA()
            RETURN 0.5 * side1 * height
        END METHOD

        OVERRIDE CALCULATE_PERIMETER()
            RETURN side1 + side2 + side3
        END METHOD
END CLASS


CLASS Rectangle INHERITS Shape 
    ATTRIBUTES
        width : NUMBER
        height : NUMBER

    CONSTRUCTOR(c : TEXT, p : TEXT, w : NUMBER, h : NUMBER)
        SUPER(c, p)
        SET width TO w
        SET height TO h
    END CONSTRUCTOR

    METHODS
        OVERRIDE CALCULATE_AREA()
            RETURN width * height
        END METHOD

        OVERRIDE CALCULATE_PERIMETER()
            RETURN 2 * (width + height)
        END METHOD
END CLASS



CLASS Canvas
    ATTRIBUTES
        shapes : LIST OF Shape

    CONSTRUCTOR()
        SET shapes TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_SHAPE(s : Shape)
            ADD s TO shapes
        END METHOD

        VIEW_SHAPES()
            SET i TO 0  
            REPEAT WHILE i LESS THAN SIZE OF shapes
                SET shapeNumber TO i + 1
                DISPLAY "Shape " + shapeNumber + ": " + shapes[i].position + ", " + shapes[i].color
                DISPLAY "Area = " + shapes[i].CALCULATE_AREA()
                DISPLAY "Perimeter = " + shapes[i].CALCULATE_PERIMETER()
                SET i TO i + 1
            END REPEAT
        END METHOD
END CLASS

//Task 296: Online Payment Methods

CLASS PaymentMethod
    ATTRIBUTES
        amount : NUMBER
        status : TEXT   

    CONSTRUCTOR(a : NUMBER)
        SET amount TO a
        SET status TO "pending"
    END CONSTRUCTOR

    METHODS
        GET_AMOUNT()
            RETURN amount
        END METHOD

        GET_STATUS()
            RETURN status
        END METHOD

        PROCESS_PAYMENT()
            DISPLAY " payment processing"
            RETURN FALSE
        END METHOD
END CLASS


CLASS CreditCardPayment INHERITS PaymentMethod
    ATTRIBUTES
        cardNumber : TEXT
        expiryDate : TEXT

    CONSTRUCTOR(a : NUMBER, cn : TEXT, ed : TEXT)
        SUPER(a)
        SET cardNumber TO cn
        SET expiryDate TO ed
    END CONSTRUCTOR

    METHODS
        OVERRIDE PROCESS_PAYMENT()
            IF LENGTH(cardNumber) = 16 AND expiryDate MATCHES "MM/YY" THEN
                SET status TO "completed"
                DISPLAY "Credit card payment approved for " + amount
                RETURN TRUE
            ELSE
                DISPLAY "Credit card validation failed"
                RETURN FALSE
            END IF
        END METHOD
END CLASS



CLASS BankTransferPayment INHERITS PaymentMethod
    ATTRIBUTES
        accountNumber : TEXT

    CONSTRUCTOR(a : NUMBER, acc : TEXT)
        SUPER(a)
        SET accountNumber TO acc
    END CONSTRUCTOR

    METHODS
        OVERRIDE PROCESS_PAYMENT()
            IF accountNumber MATCHES "ACC[0-9]{6,}" THEN
                DISPLAY "Processing bank transfer... please wait"
                SET status TO "completed"
                DISPLAY "Bank transfer completed for " + amount
                RETURN TRUE
            ELSE
                DISPLAY "Invalid account number format"
                RETURN FALSE
            END IF
        END METHOD
END CLASS


CLASS WalletPayment INHERITS PaymentMethod
    ATTRIBUTES
        walletBalance : NUMBER

    CONSTRUCTOR(a : NUMBER, balance : NUMBER)
        SUPER(a)
        SET walletBalance TO balance
    END CONSTRUCTOR

    METHODS
        OVERRIDE PROCESS_PAYMENT()
            IF walletBalance GREATER THAN OR EQUAL TO amount THEN
                SET walletBalance TO walletBalance - amount
                SET status TO "completed"
                DISPLAY "Wallet payment completed for " + amount
                RETURN TRUE
            ELSE
                DISPLAY "Insufficient wallet balance"
                RETURN FALSE
            END IF
        END METHOD
END CLASS

CLASS Order
    ATTRIBUTES
        id : TEXT
        items : LIST OF Item
        payment : PaymentMethod
        status : TEXT

    CONSTRUCTOR(oid : TEXT)
        SET id TO oid
        SET items TO LIST []
        SET payment TO NULL
        SET status TO "pending"
    END CONSTRUCTOR

    METHODS
        ADD_ITEM(i : Item)
            ADD i TO items
        END METHOD

        CALCULATE_TOTAL()
            SET total TO 0
            REPEAT FROM i = 0 TO SIZE OF items - 1
                SET total TO total + items[i].GET_PRICE()
            END REPEAT
            RETURN total
        END METHOD

        CHECKOUT(pm : PaymentMethod)
            SET payment TO pm
            DISPLAY "Checkout total: " + CALCULATE_TOTAL()
            IF payment.PROCESS_PAYMENT() = TRUE THEN
                SET status TO "paid"
                DISPLAY "Order " + id + " payment successful"
            ELSE
                SET status TO "failed"
                DISPLAY "Order " + id + " payment failed"
            END IF
        END METHOD

        GET_STATUS()
            RETURN status
        END METHOD
END CLASS

        OVERRIDE AUTHORISE_PURCHASE(amount : NUMBER)
            IF status = "blocked" THEN
                DISPLAY "Card is blocked"
                RETURN FALSE
            ELSE IF amount <= accountBalance THEN
                SET accountBalance TO accountBalance - amount
                DISPLAY "Debit purchase approved for " + amount
                RETURN TRUE
            ELSE
                DISPLAY "Insufficient balance"
                RETURN FALSE
            END IF
        END METHOD
END CLASS


CLASS CreditCard INHERITS Card
    ATTRIBUTES
        creditLimit : NUMBER
        outstandingBalance : NUMBER

    CONSTRUCTOR(id : TEXT, cn : TEXT, ed : TEXT, s : TEXT, c : Chip, cl : NUMBER)
        SUPER(id, cn, ed, s, c)
        SET creditLimit TO cl
        SET outstandingBalance TO 0
    END CONSTRUCTOR

    METHODS
        OVERRIDE AUTHORISE_PURCHASE(amount : NUMBER)
            IF status = "blocked" THEN
                DISPLAY "Card is blocked"
                RETURN FALSE
            END IF
            IF chip.VALIDATE_SECURITY() = TRUE AND (outstandingBalance + amount) <= creditLimit THEN
                SET outstandingBalance TO outstandingBalance + amount
                DISPLAY "Credit purchase approved for " + amount
                RETURN TRUE
            ELSE
                DISPLAY "Credit limit exceeded"
                RETURN FALSE
            END IF
        END METHOD

        GET_OUTSTANDING_BALANCE()
            RETURN outstandingBalance
        END METHOD
END CLASS

CLASS Device
    ATTRIBUTES
        deviceID : TEXT
        name : TEXT
        roomLocation : TEXT
        status : TEXT

    CONSTRUCTOR(id : TEXT, n : TEXT, r : TEXT, st : TEXT)
        SET deviceID TO id
        SET name TO n
        SET roomLocation TO r
        SET status TO st
    END CONSTRUCTOR

    METHODS
        GET_ID()
            RETURN deviceID
        END METHOD

        TURN_ON()
            SET status TO "on"
        END METHOD

        TURN_OFF()
            SET status TO "off"
        END METHOD

        GET_STATUS()
            RETURN status
        END METHOD

        GET_POWER_USAGE()
            RETURN 0
        END METHOD
END CLASS

CLASS SmartLight INHERITS Device
    ATTRIBUTES
        brightness : NUMBER

    CONSTRUCTOR(id : TEXT, n : TEXT, r : TEXT, st : TEXT, b : NUMBER)
        SUPER(id, n, r, st)
        SET brightness TO b
    END CONSTRUCTOR

    METHODS
        OVERRIDE GET_POWER_USAGE()
            IF status = "on" THEN
                RETURN brightness * 0.1
            ELSE
                RETURN 0
            END IF
        END METHOD
END CLASS

CLASS SmartThermostat INHERITS Device
    ATTRIBUTES
        targetTemperature : NUMBER
        currentTemperature : NUMBER

    CONSTRUCTOR(n : TEXT, r : TEXT, st : TEXT, target : NUMBER, current : NUMBER)
        SUPER(n, r, st)
        SET targetTemperature TO target
        SET currentTemperature TO current
    END CONSTRUCTOR

    METHODS
        OVERRIDE GET_POWER_USAGE()
            IF status EQUALS "on" THEN
                SET difference TO ABS(targetTemperature - currentTemperature)
                RETURN difference * 0.5
            ELSE
                RETURN 0
            END IF
        END METHOD
END CLASS


CLASS StorageUnit
    ATTRIBUTES
        capacity : NUMBER
        usedSpace : NUMBER

    CONSTRUCTOR(cap : NUMBER)
        SET capacity TO cap
        SET usedSpace TO 0
    END CONSTRUCTOR

    METHODS
        ADD_RECORDING(size : NUMBER)
            IF usedSpace + size LESS THAN OR EQUAL TO capacity THEN
                SET usedSpace TO usedSpace + size
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        GET_USED_SPACE()
            RETURN usedSpace
        END METHOD
END CLASS

CLASS SmartCamera INHERITS Device
    ATTRIBUTES
        recording : BOOLEAN
        storage : StorageUnit
        recordingRate : NUMBER   // MB per recording session

    CONSTRUCTOR(id : TEXT, n : TEXT, r : TEXT, st : TEXT, storageObj : StorageUnit, rr : NUMBER)
        SUPER(id, n, r, st)
        SET recording TO FALSE
        SET storage TO storageObj
        SET recordingRate TO rr
    END CONSTRUCTOR

    METHODS
        START_RECORDING()
            IF status = "on" THEN
                IF storage.ADD_RECORDING(recordingRate) = TRUE THEN
                    SET recording TO TRUE
                ELSE
                    SET recording TO FALSE
                    DISPLAY "Not enough storage to start recording"
                END IF
            END IF
        END METHOD

        STOP_RECORDING()
            SET recording TO FALSE
        END METHOD

        OVERRIDE GET_POWER_USAGE()
            IF status = "on" AND recording = TRUE THEN
                RETURN 5
            ELSE IF status = "on" THEN
                RETURN 1
            ELSE
                RETURN 0
            END IF
        END METHOD
END CLASS


CLASS Room
    ATTRIBUTES
        name : TEXT
        devices : LIST<Device>

    CONSTRUCTOR(n : TEXT)
        SET name TO n
        SET devices TO LIST []
    END CONSTRUCTOR

    METHODS
        ADD_DEVICE(d : Device)
            ADD d TO devices
        END METHOD

        VIEW_DEVICES()
            SET i TO 0
            REPEAT WHILE i LESS THAN SIZE OF devices
                DISPLAY "Device: " + devices[i].name + 
                        " (" + devices[i].roomLocation + ") Status: " + devices[i].GET_STATUS()
                DISPLAY "Power Usage = " + devices[i].GET_POWER_USAGE()
                SET i TO i + 1
            END REPEAT
        END METHOD
END CLASS
