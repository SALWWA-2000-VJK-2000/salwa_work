CLASS Product
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE name: TEXT
        PRIVATE category: Category
        PRIVATE bins: LIST<StorageBin>
        PRIVATE vendor: Vendor

    CONSTRUCTOR(n: TEXT, c: Category, v: Vendor)
        SET name TO n
        SET category TO c
        SET vendor TO v
        SET bins TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_BIN(b: StorageBin)
            ADD b TO bins
        END METHOD

       CALCULATE_TOTAL_STOCK()
			SET total TO 0
			SET i TO 0
			REPEAT WHILE i LESS THAN SIZE OF bins
				SET total TO total + bins[i].GET_QUANTITY()
				SET i TO i + 1
			END REPEAT
			RETURN total
		END METHOD


		PROCESS_MOVEMENT(m : StockMovement)
			IF m.type EQUALS "OUT" AND m.quantity GREATER THAN CALCULATE_TOTAL_STOCK() THEN
				DISPLAY "Error: insufficient stock"
			ELSE IF m.type EQUALS "IN" THEN
				m.bin.ADD_QUANTITY(m.quantity)
			ELSE
				m.bin.REMOVE_QUANTITY(m.quantity)
			END IF
		END METHOD

END CLASS

        PRIVATE teacher: TEXT
        PRIVATE subject: TEXT
        PRIVATE students: LIST<Student>

        PRIVATE schedule: TEXT   

    CONSTRUCTOR(t : Teacher, s : Subject, st : LIST<STUDENT>, m : NUMBER, sch : TEXT)

		SET maxSize TO m
		SET students TO LIST []

	END CONSTRUCTOR


    METHODS
	    GET_id()
            DISPLAY id
        END METHOD
		
		GET_subject()

        END METHOD
		
        ADD_STUDENT(st : Student)
			IF SIZE OF students GREATER THAN OR EQUALS maxSize THEN
				DISPLAY "Error: Class is full"
			ELSE
				SET conflict TO FALSE
				SET i TO 0
				REPEAT WHILE i LESS THAN SIZE OF st.schedule
					IF st.schedule[i] EQUALS schedule THEN
						SET conflict TO TRUE
					END IF
					SET i TO i + 1
				END REPEAT

				IF conflict EQUALS TRUE THEN
					DISPLAY "Error: Student already enrolled in another class at same time"
				ELSE
					ADD st TO students
				END IF
			END IF
		END METHOD
		
        GET_STUDENTS()
            DISPLAY students
        END METHOD

CLASS Policy
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE policyholder: Policyholder
        PRIVATE coverages: LIST<Coverage>
        PRIVATE claims: LIST<Claim>

    CONSTRUCTOR(ph: Policyholder)
        SET policyholder TO ph
        SET coverages TO  LIST[]
        SET claims TO  LIST[]
    END CONSTRUCTOR

    METHODS
        ADD_COVERAGE(c: Coverage)
            ADD c TO coverages
        END METHOD

		ADD_CLAIM(cl : Claim)
			IF cl.amount LESS THAN OR EQUAL TO cl.coverage.limit THEN
				ADD cl TO claims
			ELSE
				DISPLAY "Error: Claim exceeds coverage limit"
			END IF
		END METHOD


        CALCULATE_REMAINING_COVERAGE()
			SET i TO 0
			REPEAT WHILE i LESS THAN SIZE OF coverages
				SET c TO coverages[i]
				SET used TO 0
				SET j TO 0
				REPEAT WHILE j LESS THAN SIZE OF claims
					SET cl TO claims[j]
					IF cl.coverage EQUALS c AND cl.status EQUALS "Approved" THEN
						SET used TO used + cl.amount
					END IF
					SET j TO j + 1
				END REPEAT
				DISPLAY c.name + " remaining: " + (c.limit - used)
				SET i TO i + 1
			END REPEAT
		END METHOD

END CLASS

    CONSTRUCTOR(c : Coverage, a : NUMBER, s : TEXT)

		SET amount TO a
		SET status TO s
	END CONSTRUCTOR


    METHODS
        GET_AMOUNT()
            DISPLAY amount
        END METHOD

            DISPLAY status

        UPDATE_STATUS(newStatus : TEXT)

			ELSE IF status EQUALS "Under Review" AND (newStatus EQUALS "Approved" OR newStatus EQUALS "Rejected") THEN
				SET status TO newStatus
			ELSE
				DISPLAY "Invalid status transition"
			END IF
		END METHOD

END CLASS

    ATTRIBUTES
        PRIVATE names : TEXT
        PRIVATE contactDetails : TEXT
    CONSTRUCTOR(n : TEXT, c : TEXT)
        SET names TO n
        SET contactDetails TO c
    END CONSTRUCTOR
END CLASS

CLASS Vendor
    ATTRIBUTES
        PRIVATE name : TEXT
        PRIVATE specialty : TEXT
        PRIVATE contactDetails : TEXT
        PRIVATE bookedDates : LIST<TEXT>
    CONSTRUCTOR(n : TEXT, s : TEXT, c : TEXT)
        SET name TO n
        SET specialty TO s
        SET contactDetails TO c
        SET bookedDates TO LIST []
    END CONSTRUCTOR
    METHODS
        IS_AVAILABLE(d : TEXT)
			SET i TO 0
			REPEAT WHILE i LESS THAN SIZE OF bookedDates
				IF bookedDates[i] EQUALS d THEN
					RETURN FALSE
				END IF
				SET i TO i + 1
			END REPEAT
			RETURN TRUE
		END METHOD

END CLASS

CLASS Task
    ATTRIBUTES
        PRIVATE description : TEXT
        PRIVATE cost : NUMBER
        PRIVATE status : TEXT
        PRIVATE vendor : Vendor
    CONSTRUCTOR(desc : TEXT, c : NUMBER, v : Vendor)
        SET description TO desc
        SET cost TO c
        SET vendor TO v
        SET status TO "Pending"
    END CONSTRUCTOR
    METHODS
        MARK_DONE()
            SET status TO "Done"
        END METHOD
END CLASS

CLASS Guest
    ATTRIBUTES
        PRIVATE name : TEXT
        PRIVATE contactDetails : TEXT
    CONSTRUCTOR(n : TEXT, c : TEXT)
        SET name TO n
        SET contactDetails TO c
    END CONSTRUCTOR
END CLASS

CLASS Event
    ATTRIBUTES
        PRIVATE couple : Couple
        PRIVATE tasks : LIST<Task>
        PRIVATE guests : LIST<Guest>
        PRIVATE budget : NUMBER
    CONSTRUCTOR(c : Couple, b : NUMBER)
        SET couple TO c
        SET budget TO b
        SET tasks TO LIST []
        SET guests TO LIST []
    END CONSTRUCTOR
    METHODS
        ADD_TASK(t : Task, d : TEXT)
			IF CALCULATE_SPENDING() + t.cost LESS THAN OR EQUAL TO budget THEN
				IF t.vendor.IS_AVAILABLE(d) EQUALS FALSE THEN
					DISPLAY "Error: Vendor not available"
				ELSE
					ADD t TO tasks
					t.vendor.ASSIGN_TASK(d)
					DISPLAY "Task added: " + t.description
				END IF
			ELSE
				DISPLAY "Error: Task exceeds budget"
			END IF
		END METHOD


        ADD_GUEST(g : Guest)
            ADD g TO guests
        END METHOD

        CALCULATE_COMPLETION()
			IF SIZE OF tasks EQUALS 0 THEN
				DISPLAY "No tasks recorded"
			ELSE
				SET done TO 0
				SET i TO 0
				REPEAT WHILE i LESS THAN SIZE OF tasks
					IF tasks[i].status EQUALS "Done" THEN
						SET done TO done + 1
					END IF
					SET i TO i + 1
				END REPEAT
				DISPLAY (done / SIZE OF tasks) * 100
			END IF
		END METHOD


        CALCULATE_SPENDING()
            SET total TO 0
            SET i TO 0
            REPEAT WHILE i < SIZE OF tasks
                SET total TO total + tasks[i].cost
                SET i TO i + 1
            END REPEAT
            RETURN total
        END METHOD
END CLASS

START
	BEGIN
		CREATE OBJECT couple1 OF Couple("Ali & Sara", "99887766")
		CREATE OBJECT event1 OF Event(couple1, 10000)

		CREATE OBJECT vendor1 OF Vendor("SP1", "Flowers", "112233")
		CREATE OBJECT vendor2 OF Vendor("SP2", "Music", "445566")


		CREATE OBJECT task1 OF Task("Flower Decoration", 3000, vendor1)
		CREATE OBJECT task2 OF Task("Live Band", 5000, vendor2)

		CALL event1.ADD_TASK(task1, "2026-07-20")
		CALL event1.ADD_TASK(task2, "2026-07-20")

		CREATE OBJECT guest1 OF Guest("Omar", "778899")
		CALL event1.ADD_GUEST(guest1)

		CALL task1.MARK_DONE()

		DISPLAY event1.CALCULATE_COMPLETION()
		DISPLAY event1.CALCULATE_SPENDING()
	END
STOP

CLASS Concert
    ATTRIBUTES
        PRIVATE artist : Artist
        PRIVATE venue : Venue
        PRIVATE date : TEXT
        PRIVATE priceTiers : LIST<PriceTier>
        PRIVATE tickets : LIST<Ticket>
    CONSTRUCTOR(a : Artist, v : Venue, d : TEXT)
		IF v.IS_AVAILABLE(d) EQUALS FALSE THEN
			DISPLAY "Error: Venue already booked"
		ELSE
			SET artist TO a
			SET venue TO v
			SET date TO d
			SET priceTiers TO LIST []
			SET tickets TO LIST []
			v.BOOK_DATE(d)
		END IF
	END CONSTRUCTOR

    METHODS
        ADD_TIER(t : PriceTier)
            ADD t TO priceTiers
        END METHOD

        SELL_TICKET(seat : TEXT, tier : PriceTier, f : Fan)
			SET i TO 0
			REPEAT WHILE i LESS THAN SIZE OF tickets
				IF tickets[i].seatNumber EQUALS seat THEN
					DISPLAY "Error: Seat already sold"
					RETURN
				END IF
				SET i TO i + 1
			END REPEAT
			CREATE OBJECT newTicket OF Ticket(seat, tier, f)
			ADD newTicket TO tickets
			DISPLAY "Ticket sold to " + f.GET_NAME()
		END METHOD


        CALCULATE_REMAINING()
            SET sold TO SIZE OF tickets
            SET totalCapacity TO 0
            SET i TO 0
            REPEAT WHILE i < SIZE OF priceTiers
                SET totalCapacity TO totalCapacity + priceTiers[i].capacity
                SET i TO i + 1
            END REPEAT
            DISPLAY totalCapacity - sold
        END METHOD
END CLASS

START
	BEGIN
		CREATE OBJECT artist1 OF ARTIST("Coldplay", "Rock")
		CREATE OBJECT venue1 OF VENUE("Grand Arena")
		CREATE OBJECT concert1 OF CONCERT(artist1, venue1, "2026-08-01")

		CREATE OBJECT tier1 OF PRICETIER("VIP", 200, 50)
		CREATE OBJECT tier2 OF PRICETIER("Regular", 100, 200)
		concert1.ADD_TIER(tier1)
		concert1.ADD_TIER(tier2)

		CREATE OBJECT fan1 OF FAN("Sara", "99887766")
		CREATE OBJECT fan2 OF FAN("Omar", "11223344")

		concert1.SELL_TICKET("A1", tier1, fan1)
		concert1.SELL_TICKET("A1", tier1, fan2)

		DISPLAY concert1.CALCULATE_REMAINING()
	END
STOP


//Task 286: Software Sprint 
CLASS Developer
    ATTRIBUTES
        PRIVATE name : TEXT
        PRIVATE capacity : NUMBER
        PRIVATE assignedTasks : LIST<Task>
    CONSTRUCTOR(n : TEXT, c : NUMBER)
        SET name TO n
        SET capacity TO c
        SET assignedTasks TO LIST []
    END CONSTRUCTOR
    METHODS
        IS_AVAILABLE()
			IF SIZE OF assignedTasks LESS THAN capacity THEN
				RETURN TRUE
			ELSE
				RETURN FALSE
			END IF
		END METHOD

        ASSIGN_TASK(t : Task)
            IF IS_AVAILABLE() EQUALS TO  "AVAILABLE" THEN
                ADD t TO assignedTasks
                DISPLAY "Task assigned to " + name
            ELSE
                DISPLAY "Error: Developer at capacity"
            END IF
        END METHOD
END CLASS

CLASS Task
    ATTRIBUTES
        PRIVATE description : TEXT
        PRIVATE status : TEXT
        PRIVATE developer : Developer
    CONSTRUCTOR(d : TEXT, dev : Developer)
        SET description TO d
        SET status TO "To Do"
        SET developer TO dev
    END CONSTRUCTOR
    METHODS
        UPDATE_STATUS(newStatus : TEXT)
			IF status EQUALS "To Do" AND newStatus EQUALS "In Progress" THEN
				SET status TO newStatus
			ELSE IF status EQUALS "In Progress" AND newStatus EQUALS "Review" THEN
				SET status TO newStatus
			ELSE IF status EQUALS "Review" AND newStatus EQUALS "Complete" THEN
				SET status TO newStatus
			ELSE
				DISPLAY "Invalid status transition"
			END IF
		END METHOD

        MARK_COMPLETE()
            IF status EQUALS "Review" THEN
                SET status TO "Complete"
            ELSE
                DISPLAY "Error: Task not ready for completion"
            END IF
        END METHOD
END CLASS

CLASS Sprint
    ATTRIBUTES
        PRIVATE tasks : LIST<Task>
        PRIVATE startDate : TEXT
        PRIVATE endDate : TEXT
    CONSTRUCTOR(s : TEXT, e : TEXT)
        SET startDate TO s
        SET endDate TO e
        SET tasks TO LIST []
    END CONSTRUCTOR
    METHODS
        ADD_TASK(t : Task)
			IF t.developer.IS_AVAILABLE() EQUALS TRUE THEN
				ADD t TO tasks
				t.developer.ASSIGN_TASK(t)
			ELSE
				DISPLAY "Error: Developer cannot take more tasks"
			END IF
		END METHOD

        CALCULATE_PROGRESS()
            SET done TO 0
            SET i TO 0
            REPEAT WHILE i < SIZE OF tasks
                IF tasks[i].status EQUALS "Complete" THEN
                    SET done TO done + 1
                END IF
                SET i TO i + 1
            END REPEAT
            DISPLAY (done / SIZE OF tasks) * 100
        END METHOD
END CLASS

CLASS Backlog
    ATTRIBUTES
        PRIVATE tasks : LIST<Task>
    CONSTRUCTOR()
        SET tasks TO LIST []
    END CONSTRUCTOR
    METHODS
        ADD_TASK(t : Task)
            ADD t TO tasks
        END METHOD
        REMOVE_TASK(t : Task)
            REMOVE t FROM tasks
        END METHOD
END CLASS

START
	BEGIN
		CREATE OBJECT dev1 OF DEVELOPER("Ali", 2)
		CREATE OBJECT dev2 OF DEVELOPER("Sara", 1)

		CREATE OBJECT sprint1 OF SPRINT("2026-07-05", "2026-07-20")

		CREATE OBJECT task1 OF TASK("task1", dev1)
		CREATE OBJECT task2 OF TASK("task2", dev1)
		CREATE OBJECT task3 OF TASK("task3", dev2)

		sprint1.ADD_TASK(task1)
		sprint1.ADD_TASK(task2)
		sprint1.ADD_TASK(task3)

		task1.UPDATE_STATUS("In Progress")
		task1.UPDATE_STATUS("Review")
		task1.MARK_COMPLETE()

		DISPLAY sprint1.CALCULATE_PROGRESS()
	END
STOP



//Task 287: Farm Management 
CLASS Crop
    ATTRIBUTES
        PRIVATE name : TEXT
        PRIVATE expectedYield : NUMBER
        PRIVATE waterNeed : NUMBER
    CONSTRUCTOR(n : TEXT, y : NUMBER, w : NUMBER)
        SET name TO n
        SET expectedYield TO y
        SET waterNeed TO w
    END CONSTRUCTOR
    METHODS
        GET_EXPECTED_YIELD()
            RETURN expectedYield
        END METHOD
        GET_WATER_NEED()
            RETURN waterNeed
        END METHOD
END CLASS

CLASS Worker
    ATTRIBUTES
        PRIVATE name : TEXT
        PRIVATE dailyLimit : NUMBER
        PRIVATE assignedActivities : LIST<Activity>
    CONSTRUCTOR(n : TEXT, l : NUMBER)
        SET name TO n
        SET dailyLimit TO l
        SET assignedActivities TO LIST []
    END CONSTRUCTOR
    METHODS
        IS_AVAILABLE(d : TEXT)
            SET count TO 0
            SET i TO 0
            REPEAT WHILE i LESS THAN SIZE OF assignedActivities
                IF assignedActivities[i].date EQUALS d THEN
                    SET count TO count + 1
                END IF
                SET i TO i + 1
            END REPEAT
            IF count LESS THAN dailyLimit THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
        ASSIGN_ACTIVITY(a : Activity)
            ADD a TO assignedActivities
        END METHOD
END CLASS

CLASS Activity
    ATTRIBUTES
        PRIVATE type : TEXT
        PRIVATE date : TEXT
        PRIVATE worker : Worker
    CONSTRUCTOR(t : TEXT, d : TEXT, w : Worker)
        SET type TO t
        SET date TO d
        SET worker TO w
    END CONSTRUCTOR
    METHODS
        DESCRIBE()
            DISPLAY type + " on " + date + " by " + worker.name
        END METHOD
END CLASS

CLASS Field
    ATTRIBUTES
        PRIVATE crop : Crop
        PRIVATE activities : LIST<Activity>
    CONSTRUCTOR()
        SET activities TO LIST []
    END CONSTRUCTOR
    METHODS
        PLANT_CROP(c : Crop)
            SET crop TO c
        END METHOD
        ADD_ACTIVITY(a : Activity)
			IF a.type EQUALS "Watering" AND crop.GET_WATER_NEED() LESS THAN OR EQUAL TO 0 THEN
				DISPLAY "Error: Crop does not need watering"
			ELSE IF a.worker.IS_AVAILABLE(a.date) EQUALS FALSE THEN
				DISPLAY "Error: Worker at daily limit"
			ELSE
				ADD a TO activities
				a.worker.ASSIGN_ACTIVITY(a)
				DISPLAY "Activity logged: " + a.type
			END IF
		END METHOD

        CALCULATE_YIELD()
			IF SIZE OF activities EQUALS 0 THEN
				RETURN 0
			ELSE
				SET properCare TO 0
				SET i TO 0
				REPEAT WHILE i LESS THAN SIZE OF activities
					IF activities[i].type EQUALS "Watering" OR activities[i].type EQUALS "Fertilizing" THEN
						SET properCare TO properCare + 1
					END IF
					SET i TO i + 1
				END REPEAT
				RETURN crop.GET_EXPECTED_YIELD() * (properCare / SIZE OF activities)
			END IF
		END METHOD

END CLASS

CLASS Farm
    ATTRIBUTES
        PRIVATE fields : LIST<Field>
        PRIVATE workers : LIST<Worker>
    CONSTRUCTOR()
        SET fields TO LIST []
        SET workers TO LIST []
    END CONSTRUCTOR
    METHODS
        ADD_FIELD(f : Field)
            ADD f TO fields
        END METHOD
        ADD_WORKER(w : Worker)
            ADD w TO workers
        END METHOD
END CLASS

START
	BEGIN
		CREATE OBJECT farm1 OF FARM()

		CREATE OBJECT crop1 OF CROP("Wheat", 1000, 2)
		CREATE OBJECT field1 OF FIELD()
		field1.PLANT_CROP(crop1)
		farm1.ADD_FIELD(field1)

		CREATE OBJECT worker1 OF WORKER("Ali", 2)
		farm1.ADD_WORKER(worker1)

		CREATE OBJECT activity1 OF ACTIVITY("Watering", "2026-07-05", worker1)
		field1.ADD_ACTIVITY(activity1)

		CREATE OBJECT activity2 OF ACTIVITY("Fertilizing", "2026-07-05", worker1)
		field1.ADD_ACTIVITY(activity2)

		CREATE OBJECT activity3 OF ACTIVITY("Harvesting", "2026-07-05", worker1)
		field1.ADD_ACTIVITY(activity3)

		DISPLAY field1.CALCULATE_YIELD()
	END
STOP
