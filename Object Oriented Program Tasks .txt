CLASS Product
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE name: TEXT
        PRIVATE category: Category
        PRIVATE bins: LIST<StorageBin>
        PRIVATE vendor: Vendor

    CONSTRUCTOR(n: TEXT, c: Category, v: Vendor)
        SET name TO n
        SET category TO c
        SET vendor TO v
        SET bins TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_BIN(b: StorageBin)
            ADD b TO bins
        END METHOD

        CALCULATE_TOTAL_STOCK()
            SET total TO 0
			REPEAT WHILE i < SIZE OF bins
				SET total TO total + bins[i].GET_QUANTITY()
				SET i TO i + 1
			END REPEAT
			DISPLAY total
        END METHOD

        PROCESS_MOVEMENT(m: StockMovement)
            IF m.type EUQALS TO "OUT" AND m.quantity GREATER THAN CALCULATE_TOTAL_STOCK() THEN
                DISPLAY "Error: insufficient stock"
            ELSE
                IF m.type EUQALS TO "IN" THEN
                    m.bin.ADD_QUANTITY(m.quantity)
                ELSE
                    m.bin.REMOVE_QUANTITY(m.quantity)
                END IF
            END IF
        END METHOD
END CLASS

CLASS Policy
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE policyholder: Policyholder
        PRIVATE coverages: LIST<Coverage>
        PRIVATE claims: LIST<Claim>

    CONSTRUCTOR(ph: Policyholder)
        SET policyholder TO ph
        SET coverages TO  LIST[]
        SET claims TO  LIST[]
    END CONSTRUCTOR

    METHODS
        ADD_COVERAGE(c: Coverage)
            ADD c TO coverages
        END METHOD

        ADD_CLAIM(cl: Claim)
            IF cl.amount LESS OR EUQALS TO cl.coverage.limit THEN
                ADD cl TO claims
            ELSE
                DISPLAY "Error: Claim exceeds coverage limit"
            END IF
        END METHOD

        CALCULATE_REMAINING_COVERAGE()
            SET i TO 0
			REPEAT WHILE i < SIZE OF coverages
				SET c TO coverages[i]
				SET used TO 0
				SET j TO 0
				REPEAT WHILE j < SIZE OF claims
					SET cl TO claims[j]
					IF cl.coverage EUQAL c AND cl.status EUQAL "Approved" THEN
						SET used TO used + cl.amount
					END IF
					SET j TO j + 1
				END REPEAT
				DISPLAY c.name + (c.limit - used)
				SET i TO i + 1
			END REPEAT
		END METHOD
END CLASS

CLASS Claim
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE coverage: LIST<Coverage>
        PRIVATE amount: NUMBER
        PRIVATE status: TEXT  
    CONSTRUCTOR(c: Coverage, a: NUMBER, s:TEXT)
        SET coverage TO c
        SET amount TO a
        SET status TO LIST[]
		 
    END CONSTRUCTOR

    METHODS
        GET_AMOUNT()
            DISPLAY amount
        END METHOD

        GET_STATUS()
            DISPLAY status
        END METHOD

        UPDATE_STATUS(newStatus: TEXT)
            IF status EUQAL TO "Filed" AND newStatus EUQAL TO "Under Review" THEN
                SET status TO newStatus
            ELSE IF status EUQAL TO "Under Review" AND (newStatus EUQAL TO "Approved" OR newStatus EUQAL TO "Rejected") THEN
                SET status TO newStatus
            ELSE
                DISPLAY "Invalid status transition"
            END IF
        END METHOD
END CLASS

CLASS Event
    ATTRIBUTES
        PRIVATE couple : Couple
        PRIVATE tasks : LIST<Task>
        PRIVATE guests : LIST<Guest>
        PRIVATE budget : NUMBER
    CONSTRUCTOR(c : Couple, b : NUMBER)
        SET couple TO c
        SET budget TO b
        SET tasks TO LIST []
        SET guests TO LIST []
    END CONSTRUCTOR
    METHODS
        ADD_TASK(t : Task, d : TEXT)
            IF CALCULATE_SPENDING() + t.cost LESS THAN budget THEN
                DISPLAY "Error: Task exceeds budget"
            ELSE IF t.vendor.IS_AVAILABLE(d) EUQALS TO FALSE THEN
                DISPLAY "Error: Vendor not available"
            ELSE
                ADD t TO tasks
                CALL t.vendor.ASSIGN_TASK(d)
                DISPLAY "Task added: " + t.description
            END IF
        END METHOD

        ADD_GUEST(g : Guest)
            ADD g TO guests
        END METHOD

        CALCULATE_COMPLETION()
            SET done TO 0
            SET i TO 0
            REPEAT WHILE i < SIZE OF tasks
                IF tasks[i].status EQUALS TO "Done" THEN
                    SET done TO done + 1
                END IF
                SET i TO i + 1
            END REPEAT
            DISPLAY (done / SIZE OF tasks) * 100
        END METHOD

        CALCULATE_SPENDING()
            SET total TO 0
            SET i TO 0
            REPEAT WHILE i < SIZE OF tasks
                SET total TO total + tasks[i].cost
                SET i TO i + 1
            END REPEAT
            RETURN total
        END METHOD
END CLASS

CLASS Concert
    ATTRIBUTES
        PRIVATE artist : Artist
        PRIVATE venue : Venue
        PRIVATE date : TEXT
        PRIVATE priceTiers : LIST<PriceTier>
        PRIVATE tickets : LIST<Ticket>
    CONSTRUCTOR(a : Artist, v : Venue, d : TEXT)
        IF v.IS_AVAILABLE(d) EQUALS FALSE THEN
            DISPLAY "Error: Venue already booked"
        ELSE
            SET artist TO a
            SET venue TO v
            SET date TO d
            SET priceTiers TO LIST []
            SET tickets TO LIST []
            CALL v.BOOK_DATE(d)
        END IF
    END CONSTRUCTOR
    METHODS
        ADD_TIER(t : PriceTier)
            ADD t TO priceTiers
        END METHOD

        SELL_TICKET(seat : TEXT, tier : PriceTier, f : Fan)
            SET i TO 0
            REPEAT WHILE i < SIZE OF tickets
                IF tickets[i].seatNumber EQUALS TO seat THEN
                    DISPLAY "Error: Seat already sold"
                END IF
                SET i TO i + 1
            END REPEAT
            CREATE OBJECT newTicket OF Ticket(seat, tier, f)
            ADD newTicket TO tickets
            DISPLAY "Ticket sold to " + f.GET_NAME()
        END METHOD

        CALCULATE_REMAINING()
            SET sold TO SIZE OF tickets
            SET totalCapacity TO 0
            SET i TO 0
            REPEAT WHILE i < SIZE OF priceTiers
                SET totalCapacity TO totalCapacity + priceTiers[i].capacity
                SET i TO i + 1
            END REPEAT
            DISPLAY totalCapacity - sold
        END METHOD
END CLASS

        SET capacity TO c

               DISPLAY "AVAILABLE"

        END METHOD

            IF IS_AVAILABLE() EQUALS TO  "AVAILABLE" THEN
                ADD t TO assignedTasks
                DISPLAY "Task assigned to " + name
            ELSE
                DISPLAY "Error: Developer at capacity"
            END IF
        END METHOD

CLASS Task
    ATTRIBUTES
        PRIVATE description : TEXT
        PRIVATE status : TEXT
        PRIVATE developer : Developer
    CONSTRUCTOR(d : TEXT, dev : Developer)
        SET description TO d
        SET status TO "To Do"
        SET developer TO dev
    END CONSTRUCTOR
    METHODS
        UPDATE_STATUS(newStatus : TEXT)
            IF status = EQUALS "To Do" AND newStatus EQUALS "In Progress" THEN
                SET status TO newStatus
            ELSE IF status EQUALS "In Progress" AND newStatus EQUALS "Review" THEN
                SET status TO newStatus
            ELSE IF status EQUALS "Review" AND newStatus EQUALS "Complete" THEN
                SET status TO newStatus
            ELSE
                DISPLAY "Invalid status transition"
            END IF
        END METHOD
        MARK_COMPLETE()
            IF status EQUALS "Review" THEN
                SET status TO "Complete"
            ELSE
                DISPLAY "Error: Task not ready for completion"
            END IF
        END METHOD
END CLASS

        SET startDate TO s
        SET endDate TO e

                CALL t.developer.ASSIGN_TASK(t)
            ELSE
                DISPLAY "Error: Developer cannot take more tasks"
            END IF
        END METHOD
        CALCULATE_PROGRESS()
            SET done TO 0
            SET i TO 0
            REPEAT WHILE i < SIZE OF tasks
                IF tasks[i].status EQUALS "Complete" THEN
                    SET done TO done + 1
                END IF
                SET i TO i + 1
            END REPEAT
            DISPLAY (done / SIZE OF tasks) * 100
        END METHOD
END CLASS

CLASS Worker
    ATTRIBUTES
        PRIVATE name : TEXT
        PRIVATE dailyLimit : NUMBER
        PRIVATE assignedActivities : LIST<Activity>
    CONSTRUCTOR(n : TEXT, l : NUMBER)
        SET name TO n
        SET dailyLimit TO l
        SET assignedActivities TO LIST []
    END CONSTRUCTOR
    METHODS
        IS_AVAILABLE(d : TEXT)
            SET count TO 0
            SET i TO 0
            REPEAT WHILE i LESS THAN SIZE OF assignedActivities
                IF assignedActivities[i].date EQUALS d THEN
                    SET count TO count + 1
                END IF
                SET i TO i + 1
            END REPEAT
            IF count LESS THAN dailyLimit THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD
        ASSIGN_ACTIVITY(a : Activity)
            ADD a TO assignedActivities
        END METHOD
END CLASS

CLASS Activity
    ATTRIBUTES
        PRIVATE type : TEXT
        PRIVATE date : TEXT
        PRIVATE worker : Worker
    CONSTRUCTOR(t : TEXT, d : TEXT, w : Worker)
        SET type TO t
        SET date TO d
        SET worker TO w
    END CONSTRUCTOR
    METHODS
        DESCRIBE()
            DISPLAY type + " on " + date + " by " + worker.name
        END METHOD
END CLASS

CLASS Field
    ATTRIBUTES
        PRIVATE crop : Crop
        PRIVATE activities : LIST<Activity>
    CONSTRUCTOR()
        SET activities TO LIST []
    END CONSTRUCTOR
    METHODS
        PLANT_CROP(c : Crop)
            SET crop TO c
        END METHOD
        ADD_ACTIVITY(a : Activity)
            IF a.type EQUALS "Watering" AND crop.GET_WATER_NEED() LESS THAN OR EQUALS 0 THEN
                DISPLAY "Error: Crop does not need watering"
            ELSE IF a.worker.IS_AVAILABLE(a.date) EQUALS FALSE THEN
                DISPLAY "Error: Worker at daily limit"
            ELSE
                ADD a TO activities
                CALL a.worker.ASSIGN_ACTIVITY(a)
                DISPLAY "Activity logged: " + a.type
            END IF
        END METHOD
        CALCULATE_YIELD()
            SET properCare TO 0
            SET i TO 0
            REPEAT WHILE i LESS THAN SIZE OF activities
                IF activities[i].type EQUALS "Watering" OR activities[i].type EQUALS "Fertilizing" THEN
                    SET properCare TO properCare + 1
                END IF
                SET i TO i + 1
            END REPEAT
            RETURN crop.GET_EXPECTED_YIELD() * (properCare DIVIDED BY SIZE OF activities)
        END METHOD
END CLASS

CLASS Farm
    ATTRIBUTES
        PRIVATE fields : LIST<Field>
        PRIVATE workers : LIST<Worker>
    CONSTRUCTOR()
        SET fields TO LIST []
        SET workers TO LIST []
    END CONSTRUCTOR
    METHODS
        ADD_FIELD(f : Field)
            ADD f TO fields
        END METHOD
        ADD_WORKER(w : Worker)
            ADD w TO workers
        END METHOD
END CLASS
