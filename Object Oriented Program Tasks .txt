CLASS VETERINARY
	ATTRIBUTES
		PRIVATE ID: NUMBER
		PRIVATE NAME:TEXT
		PRIVATE OWNERS : LIST<OWNER>
		PRIVATE PETS: LIST<TEXT>[]
		PRIVATE VISITS: LIST<TEXT> []
		
		CONSTRUCTOR(n: TEXT, o:TEXT, p: LIST<TEXT>, v: LIST<TEXT>)
			SET NAME TO n
			SET OWNERS TO o
			SET PETS TO p
			SET VISITS TO v
		END CONSTRUCTOR
	
	METHODS
	
		GET_ID()
			DISPLAY ID
		END METHOD
	
		GET_NAME()
			DISPLAY  NAME
		END METHOD
		
		ADD_OWNER(o: OWNER)
            IF o IS NOT NULL THEN
                ADD o TO OWNERS
            ELSE
                DISPLAY "Invalid owner"
            END IF
        END METHOD

        ADD_PET(p: PET)
            IF p IS NOT NULL THEN
                ADD p TO PETS
            ELSE
                DISPLAY "Invalid pet"
            END IF
        END METHOD

        ADD_VISIT(v: VISIT)
            IF v IS NOT NULL THEN
                ADD v TO VISITS
            ELSE
                DISPLAY "Invalid visit"
            END IF
        END METHOD

END CLASS

CLASS Product
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE name: TEXT
        PRIVATE category: Category
        PRIVATE bins: LIST<StorageBin>
        PRIVATE vendor: Vendor

    CONSTRUCTOR(n: TEXT, c: Category, v: Vendor)
        SET name TO n
        SET category TO c
        SET vendor TO v
        SET bins TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_BIN(b: StorageBin)
            ADD b TO bins
        END METHOD

        CALCULATE_TOTAL_STOCK()
            SET total TO 0
			REPEAT WHILE i < SIZE OF bins
				SET total TO total + bins[i].GET_QUANTITY()
				SET i TO i + 1
			END REPEAT
			DISPLAY total
        END METHOD

        PROCESS_MOVEMENT(m: StockMovement)
            IF m.type EUQALS TO "OUT" AND m.quantity GREATER THAN CALCULATE_TOTAL_STOCK() THEN
                DISPLAY "Error: insufficient stock"
            ELSE
                IF m.type EUQALS TO "IN" THEN
                    m.bin.ADD_QUANTITY(m.quantity)
                ELSE
                    m.bin.REMOVE_QUANTITY(m.quantity)
                END IF
            END IF
        END METHOD
END CLASS


CLASS Category
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE name: TEXT

    CONSTRUCTOR(n: TEXT)
        SET name TO n
    END CONSTRUCTOR

    METHODS
        GET_NAME()

END CLASS


CLASS StorageBin
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE product: Product
        PRIVATE quantity: NUMBER

    CONSTRUCTOR(p: Product, q: NUMBER)
        SET product TO p
        SET quantity TO q
    END CONSTRUCTOR

    METHODS
        GET_QUANTITY()
            DISPLAY quantity
        END METHOD

        ADD_QUANTITY(q: NUMBER)
            SET quantity TO quantity + q
        END METHOD

        REMOVE_QUANTITY(q: NUMBER)
            IF q LESS THAN OR EUQALS quantity THEN
                SET quantity TO quantity - q
            ELSE
                DISPLAY "Error: not enough stock in bin"
            END IF
        END METHOD
END CLASS


CLASS StockMovement
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE product: Product
        PRIVATE bin: StorageBin
        PRIVATE type: TEXT   
        PRIVATE quantity: NUMBER
        PRIVATE date: TEXT

    CONSTRUCTOR(p: Product, b: StorageBin, t: TEXT, q: NUMBER, d: TEXT)
        SET product TO p
        SET bin TO b
        SET type TO t
        SET quantity TO q
        SET date TO d
    END CONSTRUCTOR

    METHODS
        GET_TYPE()
            DISPLAY type
        END METHOD

        GET_QUANTITY()
            DISPLAY quantity
        END METHOD

        GET_DATE()
            DISPLAY date
        END METHOD
END CLASS


START
    BEGIN
        CREATE OBJECT vendor1 OF Vendor("Global Supplies", "contact@global.om", 100)
        CREATE OBJECT category1 OF Category("Electronics")
        CREATE OBJECT product1 OF Product("Laptop", category1, vendor1)

        CREATE OBJECT bin1 OF StorageBin(product1, 50)
        CREATE OBJECT bin2 OF StorageBin(product1, 30)
        product1.ADD_BIN(bin1)
        product1.ADD_BIN(bin2)
        DISPLAY product1.CALCULATE_TOTAL_STOCK()  

        CREATE OBJECT movement1 OF StockMovement(product1, bin1, "OUT", 20, "2026-07-02")
        product1.PROCESS_MOVEMENT(movement1)
        DISPLAY product1.CALCULATE_TOTAL_STOCK()   

        CREATE OBJECT movement3 OF StockMovement(product1, bin1, "IN", 40, "2026-07-02")
        product1.PROCESS_MOVEMENT(movement3)
        DISPLAY product1.CALCULATE_TOTAL_STOCK()   

        vendor1.UPDATE_RELIABILITY(FALSE)        
        DISPLAY vendor1.GET_RELIABILITY()
    END
STOP

CLASS Classroom
    ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE teacher: TEXT
        PRIVATE subject: TEXT
        PRIVATE students: LIST<Student>
        PRIVATE maxSize: NUMBER
        PRIVATE schedule: TEXT   

    CONSTRUCTOR(t: Teacher, s: Subject,st: LIST<TEXT>, m: NUMBER, sch: TEXT)
        SET teacher TO t
        SET subject TO s
        SET maxSize TO m
		SET students TO LIST<TEXT> 
        SET schedule TO sch
    END CONSTRUCTOR

    METHODS
	    GET_id()
            DISPLAY id
        END METHOD
		
		GET_subject()
            DISPLAY subject
        END METHOD
		
        ADD_STUDENT(st: Student)
            IF SIZE OF students GREATER THAN OR EUQALS TO  maxSize THEN
                DISPLAY "Error: Class is full"
            ELSE IF st.schedule EUQALS TO schedule THEN
                DISPLAY "Error: Student already enrolled in another class at same time"
            ELSE
                ADD st TO students
            END IF
        END METHOD

        GET_STUDENTS()
            DISPLAY students
        END METHOD
END CLASS

CLASS Student
	ATTRIBUTES
        PRIVATE id: NUMBER
        PRIVATE name: TEXT
		PRIVATE schedule: TEXT
        PRIVATE grades: LIST<TEXT>
		
	CONSTRUCTOR(n: TEXT, sc:LIST<TEXT>)
        SET name TO n
        SET schedule TO sc
        SET grades TO LIST<TEXT>
    END CONSTRUCTOR
	
	METHODS
        ADD_GRADE(g: Grade)
            ADD g TO grades
        END METHOD
		
		CALCULATE_AVERAGE()
			SET totalMarks TO 0
            SET totalWeight TO 0
            REPEAT WHILE i < SIZE OF grades
				SET totalMarks TO totalMarks + (grades[i].score * grades[i].weight)
				SET totalWeight TO totalWeight + grades[i].weight
				SET i TO i + 1
			END REPEAT

			IF totalWeight > 0 THEN
				DISPLAY totalMarks / totalWeight
			ELSE
				DISPLAY " No grades recorded"
			END IF
		END METHOD
END CLASS
