CLASS VETERINARY
	ATTRIBUTES
		PRIVATE ID: NUMBER
		PRIVATE NAME:TEXT
		PRIVATE OWNERS : LIST<OWNER>
		PRIVATE PETS: LIST<TEXT>[]
		PRIVATE VISITS: LIST<TEXT> []
		
		CONSTRUCTOR(n: TEXT, o:TEXT, p: LIST<TEXT>, v:<TEXT>)
			SET NAME TO n
			SET OWNERS TO o
			SET PETS TO p
			SET VISITS TO v
		END CONSTRUCTOR
	
	METHODS
	
		GET_ID()
			DISPLAY ID
		END METHOD
	
		GET_NAME()
			DISPLAY  NAME
		END METHOD
		
		ADD_OWNER(o: OWNER)
            IF o IS NOT NULL THEN
                ADD o TO OWNERS
            ELSE
                DISPLAY "Invalid owner"
            END IF
        END METHOD

        ADD_PET(p: PET)
            IF p IS NOT NULL THEN
                ADD p TO PETS
            ELSE
                DISPLAY "Invalid pet"
            END IF
        END METHOD

        ADD_VISIT(v: VISIT)
            IF v IS NOT NULL THEN
                ADD v TO VISITS
            ELSE
                DISPLAY "Invalid visit"
            END IF
        END METHOD

END CLASS


CLASS OWNER
    ATTRIBUTES
        PRIVATE ID: NUMBER
        PRIVATE NAME: TEXT
        PRIVATE PHONE: TEXT
        PRIVATE PETS: LIST<TEXT>

    CONSTRUCTOR( n: TEXT, ph: TEXT, pe:LIST<TEXT> )
        SET NAME TO n
        SET PHONE TO ph
        SET PETS TO pe
    END CONSTRUCTOR

    METHODS
		GET_ID()
			DISPLAY ID
		END METHOD
	
		GET_NAME()
			DISPLAY  NAME
		END METHOD
		
		GET_PHONE()
			DISPLAY  PHONE
		END METHOD
		
        ADD_PET(p: PET)
            ADD p TO PETS
        END METHOD
END CLASS


CLASS PET
    ATTRIBUTES
        PRIVATE ID: NUMBER
        PRIVATE NAME: TEXT
        PRIVATE DATE_OF_BIRTH: STRING
        PRIVATE SPECIES: TEXT
        PRIVATE MEDICAL_HISTORY: LLIST<TEXT>

    CONSTRUCTOR(n: TEXT, dob: TEXT, s: TEXT, mh:LIST<TEXT>)
        SET NAME TO n
        SET DATE_OF_BIRTH TO dob
        SET SPECIES TO s
        SET MEDICAL_HISTORY TO mh
    END CONSTRUCTOR

    METHODS
		GET_ID()
			DISPLAY ID
		END METHOD
	
		GET_NAME()
			DISPLAY  NAME
		END METHOD
		
		GET_DATE_OF_BIRTH()
			DISPLAY  DATE_OF_BIRTH
		END METHOD
		
		GET_SPECIES()
			DISPLAY SPECIES
		END METHOD
		
		GET_MEDICAL_HISTORY()
			DISPLAY  MEDICAL_HISTORY
		END METHOD
      
        ADD_HISTORY(e: EXAMINATION)
            ADD e TO MEDICAL_HISTORY
        END METHOD

        GET_AGE()
            DISPLAY "Age "
        END METHOD
END CLASS


CLASS VISIT
    ATTRIBUTES
        PRIVATE ID: NUMBER
        PRIVATE PET: PET
        PRIVATE EXAMINATIONS: LLIST<TEXT>

    CONSTRUCTOR( p: PET, e:LLIST<TEXT>)
        SET PET TO p
        SET EXAMINATIONS TO e
    END CONSTRUCTOR

    METHODS
		GET_ID()
			DISPLAY ID
		END METHOD
	
		GET_PET()
			DISPLAY PET
		END METHOD
	
        ADD_EXAMINATION(e: EXAMINATION)
            ADD e TO EXAMINATIONS
        END METHOD

        CALCULATE_TOTAL()
            SET total TO 0
            REPEAT WHILE i < SIZE OF EXAMINATIONS
			SET TOTAL TO  TOTAL+1
			DISPLAY TOTAL	
        END METHOD
END CLASS


CLASS EXAMINATION
    ATTRIBUTES
        PRIVATE ID: NUMBER
        PRIVATE DIAGNOSIS: TEXT
        PRIVATE VITAL_SIGNS: TEXT
        PRIVATE URGENT: BOOLEAN
        PRIVATE TREATMENTS: LIST<TEXT>

    CONSTRUCTOR(d: TEXT, v: TEXT, u: BOOLEAN, t:LIST<TEXT>)
        SET DIAGNOSIS TO d
        SET VITAL_SIGNS TO v
        SET URGENT TO u
        SET TREATMENTS TO t
    END CONSTRUCTOR

    METHODS
		GET_ID()
			DISPLAY ID
		END METHOD
	
		GET_DIAGNOSIS()
			DISPLAY  DIAGNOSIS
		END METHOD
		
		GET_VITAL_SIGNS()
			DISPLAY VITAL_SIGNS
		END METHOD
	
        SET_CHECK_URGENT()
            IF VITAL_SIGNS = "OUT OF RANGE" THEN
                SET URGENT TO TRUE
            END IF
        END METHOD
		
		GET_URGENT()
			DISPLAY URGENT
		END METHOD
		
        ADD_TREATMENT(t: TREATMENT)
            ADD t TO TREATMENTS
        END METHOD
END CLASS


CLASS TREATMENT
    ATTRIBUTES
        PRIVATE ID: NUMBER
        PRIVATE NAME: TEXT
        PRIVATE COST: NUMBER

    CONSTRUCTOR(n: TEXT, c: NUMBER)
        SET NAME TO n
        SET COST TO c
    END CONSTRUCTOR

    METHODS
		GET_ID()
			DISPLAY ID
		END METHOD
	
		GET_NAME()
			DISPLAY  NAME
		END METHOD
		
        GET_COST()
            RETURN COST
        END METHOD
END CLASS
